#!/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin:/sbin:/bin:/usr/games:/usr/local/games
LOGNAME=/var/log/cron.log
SHELL=/bin/bash
PYTHONPATH=/home/<user>

01 * * * * source /lib/env_twoops/bin/activate && cd /lib/twoops && make user-refresh >> /var/log/twoops.log   # refresh users list at the top of every hour
20,50 * * * * source /lib/env_twoops/bin/activate && cd /lib/twoops && make runscript >> /var/log/twoops.log   # deletion checker twice each hour
0,5,10,15,20,25,30,35,40,45,50,55 * * * * source /lib/env_twoops/bin/activate && cd /lib/twoops/ && make heartbeat >> /var/log/twoops-heartbeat.log  # heartbeats every five minutes
50 23 * * * source /lib/env_twoops/bin/activate && cd /lib/twoops && make dump >> /var/log/twoops_export.log   # export deleted tweets to csv every day at 23:50
